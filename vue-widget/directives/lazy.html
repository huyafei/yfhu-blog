# v-lazy 图片懒加载

TIP

该指令是集成了 vue-lazyload (opens new window) 依赖。所以你可以使用该依赖指令的所有功能。

# 使用

// main.js
import Vue from "vue";
import VenWidget from "@vensst/vue-widget";
import error from "./assets/images/error.jpg";
import loading from "./assets/images/loading.gif";

// 全部引入及配置
Vue.use(VenWidget, {
  // vue-lazyload 配置
  lazy: {
    preLoad: 1.3,
    error: error,
    loading: loading,
    attempt: 1,
  },
});
// 按需引入及配置
// Vue.use(VenWidget.VueLazyload,, {

// 	error: error,
// 	loading: loading,
// 	attempt: 1,
// });
